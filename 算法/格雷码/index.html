public static String[] getGray(int n) {

    String[] graycode = new String[(int) Math.pow(2, n)];

    if(n == 1){
        graycode[0] = "0";
        graycode[1] = "1";
        return graycode;
    }

    String[] last = getGray(n - 1);

    for(int i = 0;i < last.length;i++){
        graycode[i] = "0" + last[i];
        graycode[graycode.length - i - 1] = "1" + last[i];

    }

    return graycode;
}
