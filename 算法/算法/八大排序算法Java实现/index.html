public static void merge(int a[], int first, int mid, int last, int temp[]){

    int i = first,j = mid+1;
    int k = 0;

    while(i <= mid && j<= last){
        if(a[i]<a[j])
            temp[k++] = a[i++];
        else
            temp[k++] = a[j++];
    }

    while(i <= mid)
        temp[k++] = a[i++];
    while(j <= last)
        temp[k++] = a[j++];

    for(i = 0 ; i < k ; i++)
        a[first+i] = temp[i];
}
